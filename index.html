הגדרות חיבור לענן (Supabase)

הזן את פרטי הפרויקט שלך ב-Supabase כדי לסנכרן את התמונות והמידע לענן האישי שלך, המאפשר גישה מכל מקום ובקבוצות נפרדות.

איך מגדירים את זה ב-2 דקות?

  1. פתח פרויקט חינם ב-Supabase.
  2. בסיום ההקמה, העתק את ה-Project URL וה-Anon Key מדף הבית של הפרויקט.
  3. לחץ על הלחצן למטה כדי להריץ את סקריפט ה-SQL ליצירת הטבלה והאחסון בתוך ה-SQL Editor שלהם.
SQL Setup Script 
-- 1. יצירת טבלת תצפיות וביטול אבטחת שורות (RLS) כדי לאפשר קריאה וכתיבה חופשית מהאתר
create table if not exists public.observations (
  id uuid default gen_random_uuid() primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  name text not null,
  category text not null,
  location text not null,
  notes text,
  image_url text not null
);

alter table public.observations disable row level security;

-- 2. יצירת תיקיית האחסון (Bucket) בשם 'album' אם אינה קיימת
insert into storage.buckets (id, name, public)
values ('album', 'album', true)
on conflict (id) do nothing;

-- 3. הגדרת מדיניות אבטחה (Policies) לאחסון לאפשר העלאה, קריאה ומחיקה ציבורית
alter table storage.objects enable row level security;

drop policy if exists "Allow public uploads" on storage.objects;
drop policy if exists "Allow public read access" on storage.objects;
drop policy if exists "Allow public deletes" on storage.objects;

create policy "Allow public uploads"
on storage.objects for insert
with check ( bucket_id = 'album' );

create policy "Allow public read access"
on storage.objects for select
using ( bucket_id = 'album' );

create policy "Allow public deletes"
on storage.objects for delete
using ( bucket_id = 'album' );
